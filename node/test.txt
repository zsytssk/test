"use strict";

function main() {
let start = new Date();
let duration = formatDuring(new Date() - start);
console.log(duration);
}

function formatDuring(mss) {
let result = '';
let days = parseInt(mss / (1000 _ 60 _ 60 _ 24));
if (days) {
result += days + "d ";
}
let hours = parseInt((mss % (1000 _ 60 _ 60 _ 24)) / (1000 _ 60 _ 60));
if (hours) {
result += hours + "h ";
}
let minutes = parseInt((mss % (1000 _ 60 _ 60)) / (1000 _ 60));
if (minutes) {
result += minutes + "m ";
}
let seconds = (mss % (1000 _ 60)) / 1000;
if (seconds) {
result += seconds + "s ";
}
if (!result) {
result = '0s'
}
return result;
}
main();
